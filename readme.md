### Start

```js

  pnpm install grantlibs -D 或 yarn add grantlibs -D

  // es
  import { calendar, store } from 'grantlibs'
  const year = calendar.getYear()

  // cjs
  const { calendar, store } = require('grantlibs')
  const year = calendar.getYear()

```
